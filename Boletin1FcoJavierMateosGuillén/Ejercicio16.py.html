#Escribe un programa que sea capaz de escribir los N primeros números de la sucesión de fibonacci.
print('¿Cuántos números de la sucesión de Fibonacci quieres ver?')
numero = input()
numero = int(numero)
def fibonacci(numero):
    a = 0
    b = 1
    aux = 0
    while aux < numero:
        print(a, end=' ')
        a, b = b, a+b
        aux = aux + 1
    print()
fibonacci(numero)
