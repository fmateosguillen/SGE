#Realiza la misma operación del ejercicio anterior pero con números
listaNumeros = []

print('Escriba tantos numeros como quiera '
      '\n(Pulse la tecla INTRO para añadirlos a la lista).'
      '\nPara terminar escriba "salir"')
def introducirNumeros():
    comprobacion = True
    while comprobacion:
        entrada = input()
        try:
            entrada = int(entrada)
            listaNumeros.append(entrada)
        except ValueError:
            if entrada != 'salir':
                print("La entrada es incorrecta: escribe un numero entero")
            else:
                print(f'Los números introducidos son: {listaNumeros}')
                comprobacion = False
introducirNumeros()
