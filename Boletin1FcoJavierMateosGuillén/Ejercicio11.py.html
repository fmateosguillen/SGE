#Escribe un programa que pida primero un número entero y después
# pida números enteros hasta que la suma de los números introducidos
# coincida con el número inicial. El programa termina escribiendo la
# lista de números.
print('Este programa te pide un número y después que introduzcas números\n'
      'hasta que su suma sea el número inicial\n')
listaNumeros = []
suma = 0
print('Indica el número inicial')
numeroInicial = input()
numeroInicial = int(numeroInicial)
while suma<numeroInicial:
    print('Introduzca un número')
    numeroSuma = input()
    numeroSuma = int(numeroSuma)
    if (suma + numeroSuma) > numeroInicial:
        print('Te has pasado, introduce otro')
    else:
        suma = suma + numeroSuma
        listaNumeros.append(numeroSuma)
print(f'¡Listo!. La suma de {listaNumeros} es {numeroInicial}')
