#Escribe un programa que genere un número aleatorio entre 0 y 10 y nos
# pida adivinarlo Tenemos 3 intentos.
import random
numero = random.randint(0, 10)
contador = 3
print(f'Intenta adivinar un número del 0 al 10. Tienes {contador} intentos')
entrada = input()
entrada = int(entrada)
while contador > 0:
    if numero == entrada:
        print('¡Acertaste!')
        contador = 0
    elif numero > entrada:
        contador = contador - 1
        print(f'Error, el número es mayor. Te quedan {contador} intentos')
        entrada = input()
        entrada = int(entrada)

    else:
        contador = contador - 1
        print(f'Error, el número es menor. Te quedan {contador} intentos')
        entrada = input()
        entrada = int(entrada)
