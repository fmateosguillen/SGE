#Implementar un programa en Python que pida un número
#indeterminado de cadenas de caracteres por el teclado,
#y cuando se finalice dicha introducción, muestre el listado de palabras.
listaPalabras = []
comprobacion = ""
print('Escriba tantas palabras como quiera '
      '\n(Pulse la tecla INTRO para añadirlas a la lista).'
      '\nPara terminar escriba "salir"')
while comprobacion != 'salir':
    cadena = input()
    cadena = str(cadena)
    comprobacion = cadena
    if comprobacion == 'salir':
        print('Las palabras que ha escrito son:')
        print(listaPalabras)
    else:
        listaPalabras.append(cadena)
