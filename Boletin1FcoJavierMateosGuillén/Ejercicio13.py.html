#Implementa una función que calcule el factorial de un número. Recuerda que el factorial de un número es el
#producto de todos los números desde ese número hasta 1. Por ejemplo, el factorial de 3, 3!, es 6

print('Indica un número para conocer su factorial')
numero = input()
numero = int(numero)
def factorial (numero):
    fact = 1
    i = 1
    while i <= numero:
        fact = fact * i
        i = i + 1
    print(fact)
factorial(numero)
